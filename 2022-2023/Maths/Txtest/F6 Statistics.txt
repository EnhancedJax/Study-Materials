# Random variables
sum P(X=x) = 1
E(X) = sum [x * P(X=x)]
E(X^2) = sum [x^2 * P(X=x)]
Var(X) = E(X^2) - E(X)^2

Y = aX + b
E(Y) = a E(X) + b
Var(Y) = a^2 Var(X)
