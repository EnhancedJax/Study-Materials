\section{Ratios}
\label{sec:ratios}

\begin{definition}
    {Current Ratio}
    \[
        \text{Current Ratio} = \frac{\text{Current Assets}}{\text{Current Liabilities}}
    \]
    It reflects the company's \textbf{ability to pay off its short-term debts} using its current assets.
\end{definition}
\label{def:current_ratio}
